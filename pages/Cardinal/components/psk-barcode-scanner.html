This component is useful in order to scan various types of barcodes (1D or 2D).

This component has no HTML children.

 // Controller import ContainerController from "../../cardinal/controllers/base-controllers/ContainerController.js"; export default class BarcodeController extends ContainerController{ constructor(element) { super(element); this.setModel({ scannerIsOn:false, data:"", }) this.on("toggle-scanner", () => { this.model.scannerIsOn = !this.model.scannerIsOn; }) } }
